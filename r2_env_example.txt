# Cloudflare R2 Configuration Example

# 1. Create a Cloudflare R2 bucket in your Cloudflare dashboard
# 2. Go to R2 > Buckets > Your-Bucket > Settings > API Tokens
# 3. Create a new API token with Read & Write permissions
# 4. Fill in the values below with your actual credentials

# R2 endpoint URL format: https://<account-id>.r2.cloudflarestorage.com
# Replace <account-id> with your Cloudflare account ID
R2_ENDPOINT=https://<account-id>.r2.cloudflarestorage.com

# R2 credentials from the API token you created
R2_ACCESS_KEY_ID=your_r2_access_key_id
R2_SECRET_ACCESS_KEY=your_r2_secret_access_key

# Your bucket name - this should match what you created in Cloudflare
R2_BUCKET_NAME=campuscrush-media

# R2 public URL for the bucket - if you've set up public access
# Format: https://pub-<id>.r2.dev OR your custom domain if configured
# If using a custom domain, format would be: https://your-custom-domain.com
R2_PUBLIC_URL=https://pub-<id>.r2.dev

# To test your configuration, run the migrate_to_r2.py script once media uploads are working
# This will migrate existing files to R2 storage
